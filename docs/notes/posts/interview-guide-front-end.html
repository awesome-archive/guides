先体验一下前端题目

for (var i = 0; i < 5; i++) {
  console.log(i);
}

这道题的输出是什么：1，2，3，4，5

for (var i = 0; i < 5; i++) {
  setTimeout(function() {
    console.log(i);
  }, 1000 * i);
}

这道题的输出是5。

for (var i = 0; i < 5; i++) {
  (function(i) {
    setTimeout(function() {
      console.log(i);
    }, i * 1000);
  })(i);
}
